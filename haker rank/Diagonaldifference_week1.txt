#include <stdio.h>
#include <stdlib.h> // for abs()

int diagonalDifference(int n, int arr[n][n]) {
    int primary = 0, secondary = 0;

    for (int i = 0; i < n; i++) {
        primary += arr[i][i];              // Top-left to bottom-right
        secondary += arr[i][n - i - 1];    // Top-right to bottom-left
    }

    return abs(primary - secondary);
}

int main() {
    int n;
    scanf("%d", &n);

    int arr[n][n];

    // Read matrix input
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            scanf("%d", &arr[i][j]);
        }
    }

    // Compute and print result
    int result = diagonalDifference(n, arr);
    printf("%d\n", result);

    return 0;
}

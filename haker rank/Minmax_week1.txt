#include <stdio.h>
#include <limits.h>

void miniMaxSum(long long arr[], int n) {
    long long total = 0;
    long long min = LLONG_MAX;
    long long max = LLONG_MIN;

    for (int i = 0; i < n; i++) {
        total += arr[i];
        if (arr[i] < min) min = arr[i];
        if (arr[i] > max) max = arr[i];
    }

    // Minimum sum = total - max element
    // Maximum sum = total - min element
    printf("%lld %lld\n", total - max, total - min);
}

int main() {
    long long arr[5];
    for (int i = 0; i < 5; i++) {
        scanf("%lld", &arr[i]);
    }

    miniMaxSum(arr, 5);

    return 0;
}

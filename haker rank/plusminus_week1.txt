#include <stdio.h>
#include <stdlib.h>

void plusMinus(int arr_count, int* arr) {
    int pos = 0, neg = 0, zer = 0;

    for (int i = 0; i < arr_count; i++) {
        if (arr[i] > 0) pos++;
        else if (arr[i] < 0) neg++;
        else zer++;
    }

    double total = (double)arr_count;
    printf("%.6f\n", pos / total);
    printf("%.6f\n", neg / total);
    printf("%.6f\n", zer / total);
}

int main() {
    int n;
    scanf("%d", &n);

    int* arr = malloc(n * sizeof(int));
    for (int i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
    }

    plusMinus(n, arr);

    free(arr);
    return 0;
}

#include <stdio.h>
#include <string.h>
#include <ctype.h>

void caesarCipher(char s[], int k) {
    int len = strlen(s);
    k = k % 26; // Rotation factor wraps around after 26

    for (int i = 0; i < len; i++) {
        char ch = s[i];
        
        if (isalpha(ch)) {
            char base = isupper(ch) ? 'A' : 'a';
            s[i] = ( (ch - base + k) % 26 ) + base;
        }
        // Non-alphabetic characters remain unchanged
    }
    printf("%s\n", s);
}

int main() {
    int n, k;
    scanf("%d", &n);
    
    char s[1010];
    scanf("%s", s);
    
    scanf("%d", &k);

    caesarCipher(s, k);
    return 0;
}

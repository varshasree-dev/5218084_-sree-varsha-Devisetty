#include <stdio.h>
#include <stdlib.h>

#define MAX_VAL 100

int* countingSort(int arr_count, int* arr) {
    // Allocate and initialize result array
    int* frequency = (int*)calloc(MAX_VAL, sizeof(int));
    if (frequency == NULL) {
        exit(EXIT_FAILURE); // Memory allocation failed
    }

    // Count occurrences of each value
    for (int i = 0; i < arr_count; i++) {
        frequency[arr[i]]++;
    }

    return frequency;
}

int main() {
    int n;
    scanf("%d", &n);

    int* arr = malloc(n * sizeof(int));
    if (arr == NULL) {
        exit(EXIT_FAILURE);
    }

    for (int i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
    }

    int* result = countingSort(n, arr);

    for (int i = 0; i < 100; i++) {
        printf("%d ", result[i]);
    }

    printf("\n");

    free(arr);
    free(result);
    return 0;
}

#include <stdio.h>
#include <string.h>
#include <stdlib.h>

#define MAX_STRING_LENGTH 101
#define MAX_STRINGS 1000

// Structure for storing string and its frequency
typedef struct {
    char str[MAX_STRING_LENGTH];
    int count;
} StringFrequency;

int main() {
    int n, q;
    char input[MAX_STRINGS][MAX_STRING_LENGTH];
    char queries[MAX_STRINGS][MAX_STRING_LENGTH];
    StringFrequency freqMap[MAX_STRINGS];
    int freqCount = 0;

    // Read number of input strings
    scanf("%d", &n);

    // Read input strings
    for (int i = 0; i < n; i++) {
        scanf("%s", input[i]);
    }

    // Count frequency of each string
    for (int i = 0; i < n; i++) {
        int found = 0;
        for (int j = 0; j < freqCount; j++) {
            if (strcmp(input[i], freqMap[j].str) == 0) {
                freqMap[j].count++;
                found = 1;
                break;
            }
        }
        if (!found) {
            strcpy(freqMap[freqCount].str, input[i]);
            freqMap[freqCount].count = 1;
            freqCount++;
        }
    }

    // Read number of queries
    scanf("%d", &q);

    // Read queries
    for (int i = 0; i < q; i++) {
        scanf("%s", queries[i]);
    }

    // Process each query
    for (int i = 0; i < q; i++) {
        int found = 0;
        for (int j = 0; j < freqCount; j++) {
            if (strcmp(queries[i], freqMap[j].str) == 0) {
                printf("%d\n", freqMap[j].count);
                found = 1;
                break;
            }
        }
        if (!found) {
            printf("0\n");
        }
    }

    return 0;
}

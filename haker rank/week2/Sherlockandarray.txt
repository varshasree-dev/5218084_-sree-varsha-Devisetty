#include <stdio.h>
#include <stdlib.h>

const char* balancedSums(int* arr, int n) {
    long long total_sum = 0;
    long long left_sum = 0;

    // Calculate total sum
    for (int i = 0; i < n; i++) {
        total_sum += arr[i];
    }

    // Iterate through array and check left/right sums
    for (int i = 0; i < n; i++) {
        // Right sum is total sum minus left sum minus current element
        long long right_sum = total_sum - left_sum - arr[i];
        if (left_sum == right_sum) {
            return "YES";
        }
        left_sum += arr[i];
    }
    return "NO";
}

int main() {
    int t;
    scanf("%d", &t);

    while (t--) {
        int n;
        scanf("%d", &n);
        int* arr = (int*)malloc(n * sizeof(int));
        if (!arr) {
            fprintf(stderr, "Memory allocation failed\n");
            return 1;
        }

        for (int i = 0; i < n; i++) {
            scanf("%d", &arr[i]);
        }

        printf("%s\n", balancedSums(arr, n));
        free(arr);
    }
    return 0;
}
